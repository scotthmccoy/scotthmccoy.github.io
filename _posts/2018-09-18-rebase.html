--- layout: post title: Rebase & Prune Branches git shortcuts date: '2018-09-18T11:04:00.001-07:00' author: Scott McCoy tags: modified_time: '2018-12-28T12:43:18.147-08:00' blogger_id: tag:blogger.com,1999:blog-250956833460526415.post-2921896390071004524 blogger_orig_url: https://scotthmccoy.blogspot.com/2018/09/rebase.html --- Put these in /usr/local/bin:

rebase:

echo "Fetching..."
git fetch
echo "Done."

FEATURE_BRANCH=$(git branch | grep \* | awk '{print $2}')
echo "$FEATURE_BRANCH"

git checkout develop
git pull origin develop
git checkout $FEATURE_BRANCH
git rebase develop

if ! [ $? -eq 0 ] ; then

 tput setaf 1;
echo "Rebase failed! Make appropriate code changes and then use:"
echo "git rebase --continue"
echo "Or to abort, use:"
echo "git rebase --abort"
tput setaf 0;

exit 1
fi
git push -f origin $FEATURE_BRANCH

tput setaf 2;
echo "Rebase successful!"
tput setaf 0;

prunebranches:
echo "Fetching..."
git fetch
echo "Done."

echo "Deleting merged branches off of local..."
git branch --merged origin/develop | grep -v HEAD | grep -v develop | grep -v master | grep -v jenkins | grep -v \* | awk '{print $1}' | xargs git branch -D
echo "Done."

echo "Deleting merged branches off of remote..."
git branch -r --merged origin/develop | grep -v HEAD | grep -v develop | grep -v master | grep -v jenkins | awk -F'/' '{print $2}' | xargs git push origin --delete
echo "Done."

echo "Fetching..."
git fetch
echo "Done."
